/* ========================================================================
   FILE 1) server/ga-db.ts
   - SQLite db + safe migrations (adds columns if missing)
   - Creates ga_activity_log
   - Assumes better-sqlite3 style API: db.exec(), db.prepare().get/all/run
   ======================================================================== */

import Database from "better-sqlite3";
import path from "node:path";
import fs from "node:fs";

const dataDir = path.resolve(process.cwd(), "data");
if (!fs.existsSync(dataDir)) fs.mkdirSync(dataDir, { recursive: true });

const dbPath = path.join(dataDir, "ga.sqlite");
export const gaDb = new Database(dbPath);

// Pragmas for stability
gaDb.pragma("journal_mode = WAL");
gaDb.pragma("foreign_keys = ON");

type ColInfo = { name: string };

function tableExists(table: string): boolean {
  const row = gaDb
    .prepare(`SELECT name FROM sqlite_master WHERE type='table' AND name=?`)
    .get(table);
  return !!row;
}

function getColumns(table: string): Set<string> {
  const cols = gaDb.prepare(`PRAGMA table_info(${table})`).all() as ColInfo[];
  return new Set(cols.map((c) => c.name));
}

function addColumnIfMissing(table: string, col: string, ddl: string) {
  const cols = getColumns(table);
  if (!cols.has(col)) {
    gaDb.exec(`ALTER TABLE ${table} ADD COLUMN ${col} ${ddl}`);
  }
}

export function ensureGaSchema() {
  // If your GA loads table name differs, update ONLY this const.
  const loadsTable = "ga_loads";

  // If table does not exist, create a minimal compatible one.
  // If you already have ga_loads created elsewhere, this won't run.
  if (!tableExists(loadsTable)) {
    gaDb.exec(`
      CREATE TABLE IF NOT EXISTS ${loadsTable} (
        id TEXT PRIMARY KEY,
        created_at TEXT DEFAULT (datetime('now')),
        status TEXT DEFAULT 'new',
        origin_city TEXT,
        origin_state TEXT,
        dest_city TEXT,
        dest_state TEXT,
        miles REAL,
        rate REAL,
        rpm REAL,
        score INTEGER,
        raw_json TEXT
      );
    `);
  }

  // Add requested columns safely
  addColumnIfMissing(loadsTable, "offered_at", "TEXT");
  addColumnIfMissing(loadsTable, "booked_at", "TEXT");
  addColumnIfMissing(loadsTable, "offered_rate", "REAL");
  addColumnIfMissing(loadsTable, "booked_rate", "REAL");
  addColumnIfMissing(loadsTable, "assigned_truck_id", "TEXT");
  addColumnIfMissing(loadsTable, "assigned_driver_id", "TEXT");
  addColumnIfMissing(loadsTable, "override_reason", "TEXT");
  addColumnIfMissing(loadsTable, "ratecon_path", "TEXT");
  addColumnIfMissing(loadsTable, "skip_reason", "TEXT"); // helpful for UI + audit

  // Activity log table
  gaDb.exec(`
    CREATE TABLE IF NOT EXISTS ga_activity_log (
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      load_id TEXT NOT NULL,
      action TEXT NOT NULL,
      message TEXT,
      meta_json TEXT,
      created_at TEXT DEFAULT (datetime('now')),
      FOREIGN KEY(load_id) REFERENCES ${loadsTable}(id) ON DELETE CASCADE
    );

    CREATE INDEX IF NOT EXISTS idx_ga_activity_load_id ON ga_activity_log(load_id);
    CREATE INDEX IF NOT EXISTS idx_ga_activity_created_at ON ga_activity_log(created_at);
  `);
}

export function gaLog(loadId: string, action: string, message?: string, meta?: unknown) {
  const metaJson = meta ? JSON.stringify(meta) : null;
  gaDb
    .prepare(
      `INSERT INTO ga_activity_log (load_id, action, message, meta_json) VALUES (?, ?, ?, ?)`
    )
    .run(loadId, action, message ?? null, metaJson);
}

// Run schema on import (safe/idempotent)
ensureGaSchema();

import express from "express";
import fs from "node:fs";
import path from "node:path";
import { gaDb, gaLog, ensureGaSchema } from "./ga-db";

ensureGaSchema();

const router = express.Router();

const LOADS_TABLE = "ga_loads";

/** Utility: basic fetch to internal Fleet API */
async function getDispatchGate(truckId: string) {
  // Prefer explicit base; fallback to local Replit single-server port 5000
  const base =
    process.env.INTERNAL_API_BASE_URL ||
    process.env.API_BASE_URL ||
    "http://localhost:5000";

  const url = `${base}/api/fleet/trucks/${encodeURIComponent(truckId)}/dispatch-gate`;
  const res = await fetch(url, { method: "GET" });
  if (!res.ok) {
    const txt = await res.text().catch(() => "");
    throw new Error(`Dispatch gate lookup failed (${res.status}): ${txt}`);
  }
  return (await res.json()) as {
    status: "GREEN" | "YELLOW" | "RED";
    risk?: number;
    reasons?: string[];
    override?: { active: boolean; reason?: string };
  };
}

/** Minimal list endpoint (if you already have one, keep yours) */
router.get("/loads", (req, res) => {
  const status = (req.query.status as string | undefined) ?? undefined;

  const rows =
    status && status.length
      ? gaDb
          .prepare(
            `SELECT * FROM ${LOADS_TABLE} WHERE status=? ORDER BY datetime(created_at) DESC LIMIT 500`
          )
          .all(status)
      : gaDb
          .prepare(`SELECT * FROM ${LOADS_TABLE} ORDER BY datetime(created_at) DESC LIMIT 500`)
          .all();

  res.json({ ok: true, loads: rows });
});

/** Offer */
router.post("/loads/:id/offer", express.json(), (req, res) => {
  const id = req.params.id;
  const offered_rate = Number(req.body?.offered_rate);

  if (!id) return res.status(400).json({ ok: false, error: "Missing load id" });
  if (!Number.isFinite(offered_rate) || offered_rate <= 0) {
    return res.status(400).json({ ok: false, error: "Invalid offered_rate" });
  }

  const row = gaDb.prepare(`SELECT id, status FROM ${LOADS_TABLE} WHERE id=?`).get(id) as any;
  if (!row) return res.status(404).json({ ok: false, error: "Load not found" });

  gaDb
    .prepare(
      `UPDATE ${LOADS_TABLE}
       SET status='offered', offered_rate=?, offered_at=datetime('now')
       WHERE id=?`
    )
    .run(offered_rate, id);

  gaLog(id, "OFFER", `Marked as offered at ${offered_rate}`, { offered_rate });

  return res.json({ ok: true });
});

/** Skip */
router.post("/loads/:id/skip", express.json(), (req, res) => {
  const id = req.params.id;
  const reason = String(req.body?.reason ?? "").trim();

  if (!id) return res.status(400).json({ ok: false, error: "Missing load id" });
  if (!reason) return res.status(400).json({ ok: false, error: "Missing reason" });

  const row = gaDb.prepare(`SELECT id FROM ${LOADS_TABLE} WHERE id=?`).get(id) as any;
  if (!row) return res.status(404).json({ ok: false, error: "Load not found" });

  gaDb
    .prepare(
      `UPDATE ${LOADS_TABLE}
       SET status='skipped', skip_reason=?
       WHERE id=?`
    )
    .run(reason, id);

  gaLog(id, "SKIP", reason);

  return res.json({ ok: true });
});

/** Activity log */

  return res.json({ ok: true, activity: rows });
});

/** Book (with Dispatch Gate check) */
router.post("/loads/:id/book", express.json(), async (req, res) => {
  const id = req.params.id;

  const booked_rate = Number(req.body?.booked_rate);
  const assigned_truck_id = String(req.body?.assigned_truck_id ?? "").trim();
  const assigned_driver_id = String(req.body?.assigned_driver_id ?? "").trim();
  const override_reason = String(req.body?.override_reason ?? "").trim();

  if (!id) return res.status(400).json({ ok: false, error: "Missing load id" });
  if (!Number.isFinite(booked_rate) || booked_rate <= 0) {
    return res.status(400).json({ ok: false, error: "Invalid booked_rate" });
  }
  if (!assigned_truck_id) {
    return res.status(400).json({ ok: false, error: "Missing assigned_truck_id" });
  }
  if (!assigned_driver_id) {
    return res.status(400).json({ ok: false, error: "Missing assigned_driver_id" });
  }

  const load = gaDb.prepare(`SELECT * FROM ${LOADS_TABLE} WHERE id=?`).get(id) as any;
  if (!load) return res.status(404).json({ ok: false, error: "Load not found" });

  // Dispatch gate enforcement
  let gate: Awaited<ReturnType<typeof getDispatchGate>> | null = null;
  try {
    gate = await getDispatchGate(assigned_truck_id);
  } catch (e: any) {
    // If fleet API temporarily fails, you may prefer to block booking (safe)
    gaLog(id, "DISPATCH_GATE_ERROR", e?.message ?? "Unknown dispatch gate error", {
      assigned_truck_id,
    });
    return res.status(502).json({ ok: false, error: e?.message ?? "Dispatch gate lookup failed" });
  }

  if (gate.status === "RED" && !override_reason) {
    gaLog(id, "BOOK_BLOCKED", "Dispatch Gate RED (override required)", {
      assigned_truck_id,
      gate,
    });
    return res.status(409).json({
      ok: false,
      error: "Dispatch Gate RED: booking blocked unless override_reason is provided",
      gate,
    });
  }

  if (gate.status === "YELLOW" && !override_reason) {
    gaLog(id, "BOOK_NEEDS_OVERRIDE", "Dispatch Gate YELLOW (override required)", {
      assigned_truck_id,
      gate,
    });
    return res.status(409).json({
      ok: false,
      error: "Dispatch Gate YELLOW: override_reason required",
      gate,
    });
  }

  gaDb
    .prepare(
      `UPDATE ${LOADS_TABLE}
       SET status='booked',
           booked_rate=?,
           booked_at=datetime('now'),
           assigned_truck_id=?,
           assigned_driver_id=?,
           override_reason=?
       WHERE id=?`
    )
    .run(booked_rate, assigned_truck_id, assigned_driver_id, override_reason || null, id);

  gaLog(id, "BOOK", `Booked at ${booked_rate}`, {
    booked_rate,
    assigned_truck_id,
    assigned_driver_id,
    gate_status: gate.status,
    override_reason: override_reason || null,
  });

  return res.json({ ok: true, gate });
});

/** RateCon Generation (safe stub) */
router.post("/loads/:id/ratecon/generate", express.json(), (req, res) => {
  const id = req.params.id;

  const load = gaDb.prepare(`SELECT * FROM ${LOADS_TABLE} WHERE id=?`).get(id) as any;
  if (!load) return res.status(404).json({ ok: false, error: "Load not found" });

  if (load.status !== "booked") {
    return res.status(409).json({ ok: false, error: "RateCon can only be generated for booked loads" });
  }

  // Create a placeholder file (swap with your real PDF generator later)
  const outDir = path.resolve(process.cwd(), "data", "ratecons");
  if (!fs.existsSync(outDir)) fs.mkdirSync(outDir, { recursive: true });

  const filename = `ratecon_${id}.txt`;
  const outPath = path.join(outDir, filename);

  const contents =
    `RATE CONFIRMATION (PLACEHOLDER)\n\n` +
    `Load ID: ${load.id}\n` +
    `Status: ${load.status}\n` +
    `Booked Rate: ${load.booked_rate ?? ""}\n` +
    `Offered Rate: ${load.offered_rate ?? ""}\n` +
    `Truck: ${load.assigned_truck_id ?? ""}\n` +
    `Driver: ${load.assigned_driver_id ?? ""}\n` +
    `Origin: ${load.origin_city ?? ""}, ${load.origin_state ?? ""}\n` +
    `Destination: ${load.dest_city ?? ""}, ${load.dest_state ?? ""}\n` +
    `Miles: ${load.miles ?? ""}\n` +
    `Generated At: ${new Date().toISOString()}\n`;

  fs.writeFileSync(outPath, contents, "utf8");

  // Store relative path so it’s stable across repl deploys
  const relativePath = path.relative(process.cwd(), outPath);

  gaDb
    .prepare(`UPDATE ${LOADS_TABLE} SET ratecon_path=? WHERE id=?`)
    .run(relativePath, id);

  gaLog(id, "RATECON_GENERATE", "RateCon generated (placeholder)", { ratecon_path: relativePath });

  return res.json({ ok: true, ratecon_path: relativePath });
});

/** Stats */
router.get("/stats", (req, res) => {
  const totals = gaDb
    .prepare(
      `SELECT status, COUNT(*) as count
       FROM ${LOADS_TABLE}
       GROUP BY status`
    )
    .all() as Array<{ status: string; count: number }>;

  const revenueRow = gaDb
    .prepare(
      `SELECT
          COALESCE(SUM(CASE WHEN status='booked' THEN booked_rate ELSE 0 END), 0) as booked_revenue,
          COALESCE(COUNT(CASE WHEN status='booked' THEN 1 END), 0) as booked_count
       FROM ${LOADS_TABLE}`
    )
    .get() as any;

  return res.json({
    ok: true,
    totals,
    booked_revenue: Number(revenueRow?.booked_revenue ?? 0),
    booked_count: Number(revenueRow?.booked_count ?? 0),
  });
});

export default router;


/* ========================================================================

import React, { useEffect, useMemo, useState } from "react";

type LoadRow = any;

async function api<T>(path: string, opts?: RequestInit): Promise<T> {
  const res = await fetch(path, {
    headers: { "Content-Type": "application/json", ...(opts?.headers ?? {}) },
    ...opts,
  });
  const data = await res.json().catch(() => ({}));
  if (!res.ok) throw new Error(data?.error || `Request failed (${res.status})`);
  return data as T;
}

export default function LoadsInboxPage() {
  const [loads, setLoads] = useState<LoadRow[]>([]);
  const [statusFilter, setStatusFilter] = useState<string>("");

  const [busy, setBusy] = useState<string>("");

  // Offer modal
  const [offerId, setOfferId] = useState<string>("");
  const [offeredRate, setOfferedRate] = useState<string>("");

  // Book modal
  const [bookId, setBookId] = useState<string>("");
  const [bookedRate, setBookedRate] = useState<string>("");
  const [truckId, setTruckId] = useState<string>("");
  const [driverId, setDriverId] = useState<string>("");
  const [overrideReason, setOverrideReason] = useState<string>("");

  // Skip modal
  const [skipId, setSkipId] = useState<string>("");
  const [skipReason, setSkipReason] = useState<string>("");

  // Activity
  const [activityId, setActivityId] = useState<string>("");
  const [activity, setActivity] = useState<any[]>([]);

  const statuses = useMemo(
    () => ["new", "offered", "booked", "skipped"],
    []
  );

  async function refresh() {
    const q = statusFilter ? `?status=${encodeURIComponent(statusFilter)}` : "";
    const data = await api<{ ok: boolean; loads: LoadRow[] }>(`/api/ga/loads${q}`);
    setLoads(data.loads ?? []);
  }

  useEffect(() => {
    refresh().catch(() => {});
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, [statusFilter]);

  async function doOffer() {
    if (!offerId) return;
    setBusy("offer");
    try {
      await api(`/api/ga/loads/${encodeURIComponent(offerId)}/offer`, {
        method: "POST",
        body: JSON.stringify({ offered_rate: Number(offeredRate) }),
      });
      setOfferId("");
      setOfferedRate("");
      await refresh();
    } finally {
      setBusy("");
    }
  }

  async function doSkip() {
    if (!skipId) return;
    setBusy("skip");
    try {
      await api(`/api/ga/loads/${encodeURIComponent(skipId)}/skip`, {
        method: "POST",
        body: JSON.stringify({ reason: skipReason }),
      });
      setSkipId("");
      setSkipReason("");
      await refresh();
    } finally {
      setBusy("");
    }
  }

  async function doBook() {
    if (!bookId) return;
    setBusy("book");
    try {
      await api(`/api/ga/loads/${encodeURIComponent(bookId)}/book`, {
        method: "POST",
        body: JSON.stringify({
          booked_rate: Number(bookedRate),
          assigned_truck_id: truckId,
          assigned_driver_id: driverId,
          override_reason: overrideReason || undefined,
        }),
      });
      setBookId("");
      setBookedRate("");
      setTruckId("");
      setDriverId("");
      setOverrideReason("");
      await refresh();
    } catch (e: any) {
      // If dispatch gate requires override, the backend returns 409 with a clear error message.
      alert(e?.message ?? "Booking failed");
      throw e;
    } finally {
      setBusy("");
    }
  }

  async function viewActivity(id: string) {
    setActivityId(id);
    const data = await api<{ ok: boolean; activity: any[] }>(
      `/api/ga/loads/${encodeURIComponent(id)}/activity`
    );
    setActivity(data.activity ?? []);
  }

  async function genRateCon(id: string) {
    setBusy("ratecon");
    try {
      const data = await api<{ ok: boolean; ratecon_path: string }>(
        `/api/ga/loads/${encodeURIComponent(id)}/ratecon/generate`,
        { method: "POST", body: JSON.stringify({}) }
      );
      alert(`RateCon generated: ${data.ratecon_path}`);
      await refresh();
    } finally {
      setBusy("");
    }
  }

  function badge(status: string) {
    const base = "inline-block px-2 py-1 rounded text-xs font-semibold";
    if (status === "booked") return `${base} bg-green-100 text-green-800`;
    if (status === "offered") return `${base} bg-yellow-100 text-yellow-800`;
    if (status === "skipped") return `${base} bg-gray-100 text-gray-700`;
    return `${base} bg-blue-100 text-blue-800`;
  }

  return (
    <div style={{ padding: 16, maxWidth: 1200 }}>
      <h2 style={{ marginBottom: 8 }}>GA Loads Inbox</h2>

      <div style={{ display: "flex", gap: 12, alignItems: "center", marginBottom: 12 }}>
        <label>
          Status:&nbsp;
          <select value={statusFilter} onChange={(e) => setStatusFilter(e.target.value)}>
            <option value="">All</option>
            {statuses.map((s) => (
              <option key={s} value={s}>
                {s}
              </option>
            ))}
          </select>
        </label>

        <button onClick={() => refresh()} disabled={!!busy}>
          Refresh
        </button>
      </div>

      <div style={{ border: "1px solid #ddd", borderRadius: 8, overflow: "hidden" }}>
        <table style={{ width: "100%", borderCollapse: "collapse" }}>
          <thead style={{ background: "#fafafa" }}>
            <tr>
              <th style={{ textAlign: "left", padding: 10, borderBottom: "1px solid #eee" }}>Status</th>
              <th style={{ textAlign: "left", padding: 10, borderBottom: "1px solid #eee" }}>Lane</th>
              <th style={{ textAlign: "right", padding: 10, borderBottom: "1px solid #eee" }}>Miles</th>
              <th style={{ textAlign: "right", padding: 10, borderBottom: "1px solid #eee" }}>Rate</th>
              <th style={{ textAlign: "right", padding: 10, borderBottom: "1px solid #eee" }}>RPM</th>
              <th style={{ textAlign: "right", padding: 10, borderBottom: "1px solid #eee" }}>Score</th>
              <th style={{ textAlign: "left", padding: 10, borderBottom: "1px solid #eee" }}>Actions</th>
            </tr>
          </thead>
          <tbody>
            {loads.map((l) => (
              <tr key={l.id}>
                <td style={{ padding: 10, borderBottom: "1px solid #f0f0f0" }}>
                  <span className={badge(l.status)}>{l.status}</span>
                </td>
                <td style={{ padding: 10, borderBottom: "1px solid #f0f0f0" }}>
                  {(l.origin_city || "") + ", " + (l.origin_state || "")} →{" "}
                  {(l.dest_city || "") + ", " + (l.dest_state || "")}
                </td>
                <td style={{ padding: 10, borderBottom: "1px solid #f0f0f0", textAlign: "right" }}>
                  {l.miles ?? ""}
                </td>
                <td style={{ padding: 10, borderBottom: "1px solid #f0f0f0", textAlign: "right" }}>
                  {l.booked_rate ?? l.offered_rate ?? l.rate ?? ""}
                </td>
                <td style={{ padding: 10, borderBottom: "1px solid #f0f0f0", textAlign: "right" }}>
                  {l.rpm ?? ""}
                </td>
                <td style={{ padding: 10, borderBottom: "1px solid #f0f0f0", textAlign: "right" }}>
                  {l.score ?? ""}
                </td>
                <td style={{ padding: 10, borderBottom: "1px solid #f0f0f0" }}>
                  <div style={{ display: "flex", gap: 8, flexWrap: "wrap" }}>
                    <button
                      onClick={() => {
                        setOfferId(l.id);
                        setOfferedRate(String(l.rate ?? l.offered_rate ?? ""));
                      }}
                      disabled={!!busy}
                    >
                      Offer
                    </button>
                    <button
                      onClick={() => {
                        setBookId(l.id);
                        setBookedRate(String(l.offered_rate ?? l.rate ?? ""));
                        setTruckId(String(l.assigned_truck_id ?? ""));
                        setDriverId(String(l.assigned_driver_id ?? ""));
                        setOverrideReason("");
                      }}
                      disabled={!!busy}
                    >
                      Book
                    </button>
                    <button
                      onClick={() => {
                        setSkipId(l.id);
                        setSkipReason("");
                      }}
                      disabled={!!busy}
                    >
                      Skip
                    </button>
                    <button onClick={() => viewActivity(l.id)} disabled={!!busy}>
                      History
                    </button>
                    {l.status === "booked" ? (
                      <button onClick={() => genRateCon(l.id)} disabled={busy === "ratecon"}>
                        RateCon
                      </button>
                    ) : null}
                  </div>
                </td>
              </tr>
            ))}
            {!loads.length ? (
              <tr>
                <td colSpan={7} style={{ padding: 16 }}>
                  No loads.
                </td>
              </tr>
            ) : null}
          </tbody>
        </table>
      </div>

      {/* Offer Modal (simple inline) */}
      {offerId ? (
        <div style={{ marginTop: 16, padding: 12, border: "1px solid #ddd", borderRadius: 8 }}>
          <h3 style={{ marginTop: 0 }}>Offer Load</h3>
          <div style={{ display: "flex", gap: 12, alignItems: "center" }}>

            <button onClick={doOffer} disabled={busy === "offer"}>

      {/* Book Modal */}

      ) : null}

      {/* Skip Modal */}
      {skipId ? (
        <div style={{ marginTop: 16, padding: 12, border: "1px solid #ddd", borderRadius: 8 }}>
          <h3 style={{ marginTop: 0 }}>Skip Load</h3>
          <div style={{ display: "flex", gap: 12, alignItems: "center" }}>
            <div>Load: {skipId}</div>
            <label style={{ minWidth: 420 }}>
              Reason:&nbsp;
              <input value={skipReason} onChange={(e) => setSkipReason(e.target.value)} />
            </label>
            <button onClick={doSkip} disabled={busy === "skip"}>
              Confirm Skip
            </button>
            <button onClick={() => setSkipId("")} disabled={!!busy}>
              Cancel
            </button>
          </div>
        </div>
      ) : null}

      {/* Activity Viewer */}
      {activityId ? (
        <div style={{ marginTop: 16, padding: 12, border: "1px solid #ddd", borderRadius: 8 }}>
          <div style={{ display: "flex", justifyContent: "space-between", alignItems: "center" }}>
            <h3 style={{ marginTop: 0, marginBottom: 8 }}>Activity: {activityId}</h3>
            <button onClick={() => setActivityId("")}>Close</button>
          </div>
          <div style={{ fontSize: 13 }}>
            {activity.map((a) => (
              <div key={a.id} style={{ padding: "6px 0", borderBottom: "1px solid #f0f0f0" }}>
                <div>
                  <b>{a.action}</b> — {a.message || ""} <span style={{ opacity: 0.7 }}>{a.created_at}</span>
                </div>
              </div>
            ))}
            {!activity.length ? <div>No activity yet.</div> : null}
          </div>
        </div>
      ) : null}
    </div>
  );
}
