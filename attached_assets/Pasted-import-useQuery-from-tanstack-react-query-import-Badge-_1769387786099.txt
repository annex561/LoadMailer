import { useQuery } from "@tanstack/react-query";
import { Badge } from "@/components/ui/badge";
import { Button } from "@/components/ui/button";
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { Progress } from "@/components/ui/progress";
import { Link } from "wouter";
import { MapPin, ArrowRight, Truck, Phone } from "lucide-react";

export default function ActiveLoads() {
  // 1. Fetch only ACTIVE loads (dispatched, in_transit)
  const { data: loads, isLoading } = useQuery({
    queryKey: ["/api/loads"],
  });

  // Filter client-side for now (or update API to support ?status=active)
  const activeLoads = loads?.filter((load: any) => 
    ["dispatched", "in_transit", "delivered"].includes(load.status)
  ) || [];

  if (isLoading) return <div className="p-8">Loading Active Fleet...</div>;

  return (
    <div className="p-6 space-y-6 bg-slate-50 min-h-screen">
      <div className="flex items-center justify-between">
        <div>
          <h1 className="text-3xl font-black text-slate-900 tracking-tight">Active Loads</h1>
          <p className="text-slate-500">Real-time tracking of dispatched freight.</p>
        </div>
        <Badge variant="outline" className="text-lg px-4 py-1 bg-white">
          {activeLoads.length} Trucks Rolling
        </Badge>
      </div>

      <div className="grid gap-6">
        {activeLoads.length === 0 ? (
          <div className="text-center py-20 bg-white rounded-xl border border-dashed">
            <p className="text-slate-400">No active loads. Go to "RateCon Inbox" to book one.</p>
          </div>
        ) : (
          activeLoads.map((load: any) => (
            <ActiveLoadCard key={load.id} load={load} />
          ))
        )}
      </div>
    </div>
  );
}

// --- Individual Load Card Component ---
function ActiveLoadCard({ load }: { load: any }) {
  // Calculate Progress based on the 13 SOP Steps
  const steps = load.sopProgress || {};
  const completedSteps = Object.values(steps).filter(Boolean).length;
  const progressPercent = (completedSteps / 13) * 100;
  
  // Find the "Next Step" text
  const getNextStep = () => {
    if (!steps.initialSms) return "Waiting: Send Load Details";
    if (!steps.tripMessage) return "Waiting: Driver Trip Msg";
    if (!steps.puArrived) return "Transit: Arriving at Pickup";
    if (!steps.puDocs) return "Action: Upload Pickup Docs";
    if (!steps.brokerConfirmed) return "Admin: Confirm w/ Broker";
    if (!steps.driverReleased) return "Transit: En Route to Delivery";
    if (!steps.docsToEinstein) return "Final: Upload POD/Invoice";
    return "Load Complete";
  };

  return (
    <Card className="hover:shadow-md transition-shadow border-l-4 border-l-blue-600">
      <CardContent className="p-6">
        <div className="flex flex-col md:flex-row gap-6 items-center">
          
          {/* 1. Route Info */}
          <div className="flex-1 min-w-[200px]">
            <div className="flex items-center gap-2 mb-2">
              <Badge className="bg-blue-100 text-blue-700 hover:bg-blue-200 border-0">
                #{load.loadNumber}
              </Badge>
              <span className="text-xs font-bold text-slate-400 uppercase tracking-wider">
                {load.brokerName}
              </span>
            </div>
            <div className="flex items-center gap-3 text-lg font-bold text-slate-800">
              <span>{load.originCity}</span>
              <ArrowRight className="w-5 h-5 text-slate-400" />
              <span>{load.destCity}</span>
            </div>
            <div className="text-xs text-slate-500 mt-1 flex gap-4">
              <span className="flex items-center gap-1"><Truck className="w-3 h-3"/> Driver #{load.assignedDriverId}</span>
              <span className="flex items-center gap-1"><Phone className="w-3 h-3"/> {load.brokerPhone || "No Phone"}</span>
            </div>
          </div>

          {/* 2. The Progress Bar (Visual Tracking) */}
          <div className="flex-1 w-full md:w-auto">
            <div className="flex justify-between text-xs mb-2 font-medium">
              <span className="text-blue-600">{getNextStep()}</span>
              <span className="text-slate-400">{completedSteps} / 13 Steps</span>
            </div>
            <Progress value={progressPercent} className="h-2 bg-slate-100" />
          </div>

          {/* 3. Action Button */}
          <div>
            <Link href={`/loads/${load.id}`}>
              <Button className="bg-slate-900 text-white hover:bg-slate-800">
                Manage Load
              </Button>
            </Link>
          </div>

        </div>
      </CardContent>
    </Card>
  );
}
