export default function ActiveLoads() {
  const { data: loads, isLoading } = useQuery({
    queryKey: ["/api/loads"],
  });

  // Filter for Active Loads Only (Dispatched, In Transit, Delivered)
  const activeLoads = loads?.filter((load: any) => 
    ["dispatched", "in_transit", "delivered"].includes(load.status)
  ) || [];

  if (isLoading) return <div className="p-8 text-slate-500">Loading Fleet Status...</div>;

  return (
    <div className="p-4 md:p-6 space-y-6 bg-slate-50 min-h-screen">
      {/* Header */}
      <div className="flex flex-col md:flex-row md:items-center justify-between gap-4">
        <div>
          <h1 className="text-3xl font-black text-slate-900 tracking-tight">Active Load Command</h1>
          <p className="text-slate-500">Live tracking, SOP enforcement, and driver communication.</p>
        </div>
        <div className="flex gap-2">
          <Badge variant="secondary" className="text-lg px-4 py-1">
            {activeLoads.length} Trucks Active
          </Badge>
        </div>
      </div>

      {/* Grid of Command Cards */}
      <div className="grid grid-cols-1 xl:grid-cols-2 gap-6">
        {activeLoads.length === 0 ? (
          <div className="col-span-full text-center py-20 bg-white rounded-xl border border-dashed">
            <p className="text-slate-400">No active loads. Go to <a href="/inbox" className="text-blue-600 underline">RateCon Inbox</a> to dispatch one.</p>
          </div>
        ) : (
          activeLoads.map((load: any) => (
            <LoadCommandCard key={load.id} load={load} />
          ))
        )}
      </div>
    </div>
  );
}

// --- SUB-COMPONENT: DRIVER CHAT WINDOW ---
function DriverChatWindow({ load }: { load: any }) {
  const [message, setMessage] = useState("");
  const { toast } = useToast();
  const queryClient = useQueryClient();

  // Fetch Message History (Mocked endpoint for now, or use real one)
  const { data: messages } = useQuery({
    queryKey: [`/api/messages/load/${load.id}`],
    // If endpoint doesn't exist yet, it returns empty array safely
    queryFn: async () => {
      try {
        const res = await fetch(`/api/messages/load/${load.id}`);
        if(!res.ok) return [];
        return await res.json();
      } catch (e) { return []; }
    }
  });

  const sendMessage = useMutation({
    mutationFn: async () => {
      // Use your existing SMS endpoint
      await apiRequest("POST", "/api/sms/send-template", { 
        loadId: load.id, 
        type: "CUSTOM", // We might need to handle custom text in backend
        customBody: message 
      });
    },
    onSuccess: () => {
      setMessage("");
      toast({ title: "Sent", description: "Message sent to driver." });
      // Invalidate query if you have real message history
    },
    onError: () => {
      toast({ title: "Error", variant: "destructive", description: "Failed to send SMS." });
    }
  });

  return (
    <div className="flex flex-col h-full bg-slate-50">
      {/* Message History Area */}
      <div className="flex-1 p-4 overflow-y-auto space-y-4">
        {/* Placeholder for 'No messages' */}
        <div className="flex justify-center mt-10 opacity-50">
           <div className="text-center">
             <Clock className="w-8 h-8 mx-auto mb-2 text-slate-300" />
             <p className="text-xs text-slate-400">Chat History with Driver #{load.assignedDriverId}</p>
           </div>
        </div>

        {/* Example of how messages would look */}
        {/* <div className="flex justify-end">
          <div className="bg-blue-600 text-white text-sm px-3 py-2 rounded-lg rounded-tr-none max-w-[80%]">
            Are you unloaded yet?
          </div>
        </div>
        <div className="flex justify-start">
          <div className="bg-white border text-slate-800 text-sm px-3 py-2 rounded-lg rounded-tl-none max-w-[80%]">
            Yes, waiting on paperwork.
          </div>
        </div> 
        */}
      </div>

      {/* Input Area */}
      <div className="p-3 bg-white border-t flex gap-2">
        <Input 
          placeholder="Type message to driver..." 
          className="flex-1"
          value={message}
          onChange={(e) => setMessage(e.target.value)}
          onKeyDown={(e) => e.key === "Enter" && sendMessage.mutate()}
        />
        <Button 
          size="icon" 
          className="bg-blue-600 hover:bg-blue-700"
          onClick={() => sendMessage.mutate()}
          disabled={!message || sendMessage.isPending}
        >
          <Send className="w-4 h-4" />
        </Button>
      </div>
    </div>
  );
}
