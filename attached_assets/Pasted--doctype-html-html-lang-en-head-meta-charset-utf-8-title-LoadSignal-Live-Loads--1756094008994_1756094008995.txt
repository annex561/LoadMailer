  <script>
    const bodyEl    = document.getElementById('body');
    const countEl   = document.getElementById('count');
    const updatedEl = document.getElementById('updated');
    const qEl       = document.getElementById('q');
    const equipEl   = document.getElementById('equip');
    const refreshEl = document.getElementById('refresh');

    let timer;

    function keyOf(r){
      return `${(r['Pick Up']||'').trim()}→${(r['Delivery']||'').trim()}|${(r['pick up date']||'').trim()}|${(r['Company']||'').trim()}|${(r['Contact Info']||'').trim()}`.toLowerCase();
    }

    function fmtPay(x){
      if (!x) return '';
      const n = Number(String(x).replace(/[^0-9.]/g,'')); if (!n) return x;
      return '$' + n.toLocaleString();
    }

    function render(rows){
      bodyEl.innerHTML = '';
      const q = qEl.value.trim().toLowerCase();
      const eq = equipEl.value.trim().toLowerCase();

      // client-side dedupe (server already de-dupes)
      const seen = new Set();
      const filtered = [];
      for (const r of rows) {
        const k = keyOf(r);
        if (seen.has(k)) continue;
        seen.add(k);

        const hay = [
          r['Pick Up'], r['Delivery'], r['Company'], r['Contact Info'], r['Load Type']
        ].join(' ').toLowerCase();

        if (q && !hay.includes(q)) continue;
        if (eq && String(r['Load Type']||'').toLowerCase() !== eq) continue;

        filtered.push(r);
      }

      countEl.textContent = `${filtered.length} load${filtered.length===1?'':'s'}`;

      const frag = document.createDocumentFragment();
      for (const r of filtered) {
        const tr = document.createElement('tr');
        tr.innerHTML = `
          <td class="pay">${fmtPay(r['Pay'])}</td>
          <td>${r['Total miles']||''}</td>
          <td>${r['Pick Up']||''}</td>
          <td>${r['Delivery']||''}</td>
          <td>${r['pick up date']||''}</td>
          <td>${r['Deadhead']||''}</td>
          <td>${r['Weight']||''}</td>
          <td><span class="pill">${r['Load Type']||''}</span></td>
          <td><div>${(r['Contact Info']||'').replace(/\\n/g,'<br>')}</div></td>
          <td>${r['Company']||''}</td>
        `;
        frag.appendChild(tr);
      }
      bodyEl.appendChild(frag);
      updatedEl.textContent = new Date().toLocaleTimeString();
    }

    async function load(){
      try{
        const res = await fetch('/api/loads');
        const json = await res.json();
        render(json);
      }catch(e){
        console.error(e);
      }
    }

    refreshEl.addEventListener('click', load);
    qEl.addEventListener('input', () => render(window.__rows||[]));
    equipEl.addEventListener('change', () => render(window.__rows||[]));

    async function tick(){
      const res = await fetch('/api/loads');
      const rows = await res.json();
      window.__rows = rows;
      render(rows);
    }

    // initial & auto refresh every 30s
    tick();
    timer = setInterval(tick, 30000);
  </script>
