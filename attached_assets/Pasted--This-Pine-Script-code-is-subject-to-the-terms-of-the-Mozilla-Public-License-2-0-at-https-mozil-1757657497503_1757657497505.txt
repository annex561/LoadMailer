FindBox(_timeframe) =>
    var int bearBull = na
    var float top = na
    var float bottom = na
    var int left = na
    if barstate.isconfirmed
        bearBull := low[2] >= high ? -1 : low >= high[2] ? 1 : 0
        top := bearBull > 0 ? low : bearBull < 0 ? low[2] : 0
        bottom := bearBull > 0 ? high[2] : bearBull < 0 ? high : 0
        left := time - _timeframe * 60000 * 2
    [left, top, bottom, bearBull]

CreateBox(_time, _top, _bottom, x, TF) =>
    _col = x > 0 ? i_bullColor : i_bearColor
    _extend = i_extend_r ? extend.right : extend.none
    box boxOB = na
    if x != 0
        boxOB := box.new(left=_time, top=_top, right=time, extend=_extend , bottom=_bottom, bgcolor=_col, border_color = na, xloc = xloc.bar_time)
    if x > 0
        array.push(bullBoxes, boxOB)
    if x < 0
        array.push(bearBoxes, boxOB)

UpdateBox(_boxes, _bearBull) => 
    if array.size(_boxes) > 0
        for i = array.size(_boxes) - 1 to 0 by 1
            box = array.get(_boxes, i)
            boxLow = box.get_bottom(box)
            boxHigh = box.get_top(box)
            if (_bearBull > 0 and low < boxLow) or (_bearBull < 0 and high > boxHigh)
                box.set_right(box, time)
                box.set_extend(box, extend.none)
                if i_changeColor
                    box.set_bgcolor(box, i_bullColorTested)
                array.set(_boxes, i, na)
                array.push(mitigatedBoxes, box)
            else 
                if _bearBull > 0 and low < boxHigh                
                    oldLow = box.get_bottom(box)
                    box.set_bottom(box, low)
                    box.set_extend(box, extend.none)
                    array.set(_boxes, i, na)
                    array.push(mitigatedBullBoxes, box)

                    _col =  i_bullColor
                    _extend = i_extend_r ? extend.right : extend.none
                    box boxOB = na
                    if _bearBull != 0
                        boxOB := box.new(left=box.get_left(box), top=low, right=time, extend=_extend , bottom=oldLow, bgcolor=_col, border_color = na, xloc = xloc.bar_time)
                    if _bearBull > 0
                        array.push(bullBoxes, boxOB)
                    if _bearBull < 0
                        array.push(bearBoxes, boxOB)

                    if i_changeColor
                        box.set_bgcolor(box, i_bullColorTested)
                        array.set(_boxes, i, na)

                if _bearBull < 0 and high > boxLow
                    oldHigh = box.get_top(box)
                    box.set_top(box, high)
                    box.set_extend(box, extend.none)
                    array.set(_boxes, i, na)
                    array.push(mitigatedBearBoxes, box)

                    _col =  i_bearColor
                    _extend = i_extend_r ? extend.right : extend.none
                    box boxOB = na
                    if _bearBull != 0
                        boxOB := box.new(left=box.get_left(box), top=oldHigh, right=time, extend=_extend , bottom=high, bgcolor=_col, border_color = na, xloc = xloc.bar_time)
                    if _bearBull > 0
                        array.push(bullBoxes, boxOB)
                    if _bearBull < 0
                        array.push(bearBoxes, boxOB)

                    if i_changeColor
                        box.set_bgcolor(box, i_bearColorTested)
                        array.set(_boxes, i, na)
                box.set_right(box, time)
        
    if array.size(_boxes) > 0 
        for i = array.size(_boxes) - 1 to 0 by 1
            if na(array.get(_boxes, i))
                array.remove(_boxes, i)

FindMitigatedBoxes(_mitigatedBoxes) =>
    _mitigatedaFVG = false
    if array.size(_mitigatedBoxes) > 0
        for i = array.size(_mitigatedBoxes) - 1 to 0 by 1
            if na(array.get(_mitigatedBoxes, i))
                array.remove(_mitigatedBoxes, i)
            else
                box =  array.get(_mitigatedBoxes, i)
                left = box.get_left(box)
                right = box.get_right(box)
                top = box.get_top(box)
                bottom = box.get_bottom(box)

                if low <= top and high >= bottom and right == time
                    _mitigatedaFVG := true

                // Cull bars (we don't need mitigated FVG's to take up space)
                if right < time - (60000 * i_FVGBarCull)
                    array.set(_mitigatedBoxes, i, na)
                    box.delete(box)
    _mitigatedaFVG

// enums for candle classifications
enum tSize
    zero
    tiny
    small
    long
enum tCandleSide
    upper
    lower

GetWickSize(tCandleSide side) =>    
    float wickSize = na
    if side == tCandleSide.upper
        wickSize := high - bodyHigh
    else
        wickSize := bodyLow - low

    // true: classify relative to percent of candleRange
    // false: classify relative to percent of bodyRange
    useCandleRangeSizeLogic = false

    if wickSize == 0
        tSize.zero
    else if useCandleRangeSizeLogic ? wickSize < candleRange * 0.1 :  wickSize < bodyRange * 0.2
        tSize.tiny
    else if useCandleRangeSizeLogic ? wickSize >= candleRange * 0.1 and wickSize < candleRange * 0.7 : wickSize >= bodyRange * 0.2 and wickSize < bodyRange * 1.2
        tSize.small
    else if useCandleRangeSizeLogic ? wickSize >= candleRange * 0.7 : wickSize >= bodyRange * 1.2
        tSize.long

GetBodySize() =>
    if math.round_to_mintick(close) == math.round_to_mintick(open)
        tSize.zero
    else if bodyRange < avgBodyRange * 0.25
        tSize.tiny
    else if bodyRange >= avgBodyRange * 0.25 and bodyRange < avgBodyRange * 0.7
        tSize.small
    else if bodyRange > avgBodyRange * 0.7
        tSize.long

tSize lowerWick = GetWickSize(tCandleSide.lower)
tSize upperWick = GetWickSize(tCandleSide.upper)
tSize bodySize = GetBodySize()
