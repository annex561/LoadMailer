import { useState } from "react";
import { Check, Lock, Send, Clock, AlertTriangle } from "lucide-react";
import { useMutation, useQueryClient } from "@tanstack/react-query";
import { apiRequest } from "@/lib/queryClient";
import { Button } from "@/components/ui/button";
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { Checkbox } from "@/components/ui/checkbox";
import { Badge } from "@/components/ui/badge";
import { Separator } from "@/components/ui/separator";
import { useToast } from "@/hooks/use-toast";

interface EVChecklistProps {
  load: any; 
}

export function EVChecklist({ load }: EVChecklistProps) {
  const { toast } = useToast();
  const queryClient = useQueryClient();
  
  // 1. Load Saved Progress or Default to False
  const [steps, setSteps] = useState(load.sopProgress || {
    initialSms: false,      // Step 1
    tripMessage: false,     // Step 2
    puArrived: false,       // Step 3
    annexNotified: false,   // Step 4
    puDocs: false,          // Step 5
    brokerConfirmed: false, // Step 6
    transitMonitored: false,// Step 7
    delDocsRequested: false,// Step 8
    driverReleased: false,  // Step 9
    docsToEinstein: false,  // Step 10
    factoringSent: false,   // Step 11
    fuelLogged: false,      // Step 12
    brokerThankYou: false,  // Step 13
  });

  // 2. Save Progress to DB
  const updateProgress = useMutation({
    mutationFn: async (newSteps: any) => {
      await apiRequest("PATCH", `/api/loads/${load.id}`, { sopProgress: newSteps });
    },
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: [`/api/loads/${load.id}`] });
    },
  });

  const toggleStep = (key: string) => {
    const newSteps = { ...steps, [key]: !steps[key] };
    setSteps(newSteps);
    updateProgress.mutate(newSteps);
  };

  // 3. Automated SMS Handlers
  const handleSmsTrigger = async (type: string, stepKey: string) => {
    try {
      await apiRequest("POST", `/api/sms/send-template`, { loadId: load.id, type });
      toast({ title: "SMS Sent", description: "Driver notified successfully." });
      toggleStep(stepKey); // Auto-check the box
    } catch (e) {
      toast({ title: "Error", variant: "destructive", description: "Failed to send SMS" });
    }
  };

  return (
    <Card className="h-full border-l-4 border-l-blue-600 shadow-md flex flex-col">
      <CardHeader className="bg-slate-50 pb-2">
        <CardTitle className="text-lg font-bold text-slate-800 flex justify-between">
          SOP Checklist
          <Badge variant="outline">{Object.values(steps).filter(Boolean).length} / 13</Badge>
        </CardTitle>
        <p className="text-xs text-slate-500">Strict Sequential Enforcement Active</p>
      </CardHeader>

      <CardContent className="flex-1 overflow-y-auto p-4 space-y-4">
        
        {/* PHASE 1: DISPATCH */}
        <div className="space-y-3">
          <h4 className="text-xs font-bold text-slate-400 uppercase tracking-wider">Phase 1: Dispatch</h4>
          
          {/* STEP 1: Always Enabled */}
          <div className="flex items-start gap-3">
            <Checkbox checked={steps.initialSms} onCheckedChange={() => toggleStep("initialSms")} />
            <div className="grid gap-1">
              <label className="text-sm font-medium">1. Send Load Details</label>
              <Button size="xs" variant="outline" className="h-6 text-xs w-full" 
                onClick={() => handleSmsTrigger("INITIAL", "initialSms")}
                disabled={steps.initialSms}> {/* Disable button if already done */}
                <Send className="w-3 h-3 mr-1" /> Send SMS
              </Button>
            </div>
          </div>

          {/* STEP 2: Locked until Step 1 is done */}
          <SopItem 
            label="2. Trip Message (On-Site)" 
            checked={steps.tripMessage} 
            onChange={() => toggleStep("tripMessage")} 
            locked={!steps.initialSms} 
          />
        </div>

        <Separator />

        {/* PHASE 2: TRANSIT */}
        <div className="space-y-3">
          <h4 className="text-xs font-bold text-slate-400 uppercase tracking-wider">Phase 2: Transit</h4>
          
          <SopItem label="3. Pickup Arrived" checked={steps.puArrived} onChange={() => toggleStep("puArrived")} locked={!steps.tripMessage} />
          
          <SopItem label="4. Annex Notified" checked={steps.annexNotified} onChange={() => toggleStep("annexNotified")} locked={!steps.puArrived} />
          
          <SopItem label="5. Pickup Docs (BOL/Photo)" checked={steps.puDocs} onChange={() => toggleStep("puDocs")} locked={!steps.annexNotified} />
          
          <SopItem label="6. Broker Confirmed" checked={steps.brokerConfirmed} onChange={() => toggleStep("brokerConfirmed")} locked={!steps.puDocs} />
          
          <SopItem label="7. Transit Monitored" checked={steps.transitMonitored} onChange={() => toggleStep("transitMonitored")} locked={!steps.brokerConfirmed} />
        </div>

        <Separator />

        {/* PHASE 3: COMPLETION */}
        <div className="space-y-3">
          <h4 className="text-xs font-bold text-slate-400 uppercase tracking-wider">Phase 3: Completion</h4>
          
          <SopItem label="8. Del. Docs Requested" checked={steps.delDocsRequested} onChange={() => toggleStep("delDocsRequested")} locked={!steps.transitMonitored} />
          
          <SopItem label="9. Driver Released" checked={steps.driverReleased} onChange={() => toggleStep("driverReleased")} locked={!steps.delDocsRequested} />
          
          <SopItem label="10. Docs to Einstein" checked={steps.docsToEinstein} onChange={() => toggleStep("docsToEinstein")} locked={!steps.driverReleased} />
          
          <SopItem label="11. Factoring Sent" checked={steps.factoringSent} onChange={() => toggleStep("factoringSent")} locked={!steps.docsToEinstein} />
          
          <SopItem label="12. Fuel Logged" checked={steps.fuelLogged} onChange={() => toggleStep("fuelLogged")} locked={!steps.factoringSent} />
          
          {/* STEP 13: Final Step */}
          <div className="opacity-100">
             <div className="flex items-center gap-2 mb-2">
               <Checkbox 
                 checked={steps.brokerThankYou} 
                 onCheckedChange={() => toggleStep("brokerThankYou")} 
                 disabled={!steps.fuelLogged} // Locked until Step 12
               />
               <label className={`text-sm font-medium ${!steps.fuelLogged ? "text-slate-400" : ""}`}>
                 13. Broker Thank You
               </label>
               {!steps.fuelLogged && <Lock className="w-3 h-3 text-slate-300" />}
             </div>
          </div>
        </div>
      </CardContent>
    </Card>
  );
}

// --- Helper Component for Locked Items ---
function SopItem({ label, checked, onChange, locked }: { label: string, checked: boolean, onChange: () => void, locked: boolean }) {
  return (
    <div className={`flex items-center gap-3 transition-opacity duration-200 ${locked ? "opacity-50" : "opacity-100"}`}>
      <Checkbox 
        checked={checked} 
        onCheckedChange={onChange} 
        disabled={locked} 
      />
      <label className="text-sm leading-none font-medium flex-1 cursor-pointer">
        {label}
      </label>
      {locked && <Lock className="w-3 h-3 text-slate-400" />}
    </div>
  );
}
